```{r}

  read_csv("datos-encuestas/tabla_bogota.csv")

  read_csv("datos-encuestas/tabla_cali.csv")

  read_csv("datos-encuestas/tabla_bquilla.csv")
tabla_cartagena <-

lista_tablas <- list(
